40 # Babel provides a nice wrapper around Neptune jobs. Instead of making users
 
41 # write multiple Neptune jobs to actually run code (e.g., putting input in the
 
42 # datastore, run the job, get the output back), Babel automatically handles
 
43 # this.
 
44 def babel(params)
 
45   # Since this whole function should run asynchronously, we run it as a future.
 
46   # It automatically starts running in a new thread, and attempting to get the
 
47   # value of what this returns causes it to block until the job completes.
 
48   future {
 
49     if params[:storage]
 
50       params[:is_remote] = true
 
51     else
 
52       params[:is_remote] = false
 
53     end
 
54 
55     job_data = BabelHelper.convert_from_neptune_params(params)
 
56 
57     NeptuneHelper.validate_storage_params(job_data)  # adds in S3 storage params
 
58 
59     # :code is the only required parameter - everything else can use default vals
 
60     NeptuneHelper.require_param("@code", job_data)
 
61     BabelHelper.check_output_files(job_data)
 
62 
63     if job_data["@is_remote"]
 
64       BabelHelper.validate_inputs(job_data)
 
65     else
 
66       BabelHelper.put_code(job_data)
 
67       BabelHelper.put_inputs(job_data)
 
68     end
 
69 
70     BabelHelper.run_job(job_data)
 
71 
72     # Return an object to the user that has all the information about their
 
73     # task, including its standard out, err, debugging info, and profiling
 
74     # info. Since the job may not be done when the user asks for this info,
 
75     # its the responsibility of TaskInfo objects to block until that info
 
76     # is ready. We don't explicitly return the TaskInfo object, because it's
 
77     # in a Future block - it will automatically return whatever the last
 
78     # statement returns.
 
79     TaskInfo.new(job_data)
 
80   }
